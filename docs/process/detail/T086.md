## 3. 轨道战斗

### 3.1 轨道平台攻击

```go
// ProcessOrbitalPlatformAttack 处理轨道平台攻击
func ProcessOrbitalPlatformAttack(platform *OrbitalPlatform, target *EnemyForce, currentTick int64) (damage int, success bool) {
    if platform.HP <= 0 {
        return 0, false
    }

    // 检查冷却
    if currentTick-platform.Weapon.LastFireTick < int64(platform.Weapon.FireRate) {
        return 0, false
    }

    // 检查弹药
    if platform.AmmoCount < platform.Weapon.AmmoCost {
        return 0, false
    }

    // 计算距离（轨道平台到地面目标）
    distance := calculateOrbitalDistance(platform.Orbit, target.Position)
    if distance > platform.Weapon.Range {
        return 0, false
    }

    damage = platform.Weapon.Damage
    target.Strength -= damage / 5 // 轨道攻击造成力量减少

    platform.AmmoCount -= platform.Weapon.AmmoCost
    platform.Weapon.LastFireTick = currentTick

    return damage, true
}
```

## 4. 编队系统

### 4.1 编队移动

```go
// CalculateFormationPositions 计算编队位置
func CalculateFormationPositions(leader *CombatUnit, formationType FormationType, spacing float64) []Position {
    positions := make([]Position, 0)

    switch formationType {
    case FormationTypeLine:
        for i := 0; i < 5; i++ {
            pos := Position{
                X: leader.Position.X - int(spacing*float64(i)),
                Y: leader.Position.Y,
            }
            positions = append(positions, pos)
        }
    case FormationTypeVee:
        for i := 0; i < 5; i++ {
            offset := int(spacing * float64(i))
            pos := Position{
                X: leader.Position.X - offset,
                Y: leader.Position.Y - offset,
            }
            if i%2 == 0 {
                pos.Y = leader.Position.Y + offset
            }
            positions = append(positions, pos)
        }
    // ... 其他队形
    }
    return positions
}
```
